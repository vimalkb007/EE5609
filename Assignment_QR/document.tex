\section{Problem}
Perform QR decomposition of \myvec{-1&3\\1&5} 
\section{Explanation}
Let $\vec{a}$ and $\vec{b}$ be columns of a $\vec{V}$. Then, the matrix $\vec{V}$ can be decomposed in the form as:
\begin{align}
 \vec{V} = \vec{QR}\label{eq10}
\end{align}
such that  
\begin{align}
	& \vec{Q}\vec{Q}^T= \vec{Q}^T\vec{Q} = \vec{I} \\ 
	& \vec{Q} = \myvec{\vec{u_1} & \vec{u_2}}  \\
	& \vec{R} = \myvec{k_1 & r_1 \\ 0 & k_2} 
\end{align}
where
\begin{align}
& k_1 = \norm{\vec{a}} \label{1} \\
& \vec{u_1} = \frac{\vec{a}}{k_1} \label{2} \\
& r_1 = \frac{\vec{u_1}^T\vec{b}}{\norm{\vec{u_1}}^2} \label{3} \\
& \vec{u_2} = \frac{\vec{b}-r_1\vec{u_1}}{\norm{\vec{b}-r_1\vec{u_1}}} \label{4} \\
& k_2 = \vec{u_2}^T\vec{b} \label{5}
\end{align}
Then, the matrix can be represented as 
\begin{align}
	\myvec{\vec{a} & \vec{b} } = \myvec{\vec{u_1} & \vec{u_2}}\myvec{k_1 & r_1 \\ 0 & k_2} \label{6}
\end{align}
\section{Solution}
 $\vec{V} = \myvec{-1&3\\1&5}$ where
\begin{align}
	\vec{a} = \myvec{-1\\1} \\
	\vec{b} =\myvec{3\\5}
\end{align}
Now, from \eqref{1} and \eqref{2}, we have
\begin{align}
&	k_1 = \norm{\vec{a}} = \sqrt{2}\\
&	\vec{u_1} = \frac{1}{\sqrt{2}}\myvec{-1\\1}
\end{align}
By, \eqref{3}, we find
\begin{align}
	r_1 = \frac{\frac{1}{\sqrt{2}}\myvec{-1 & 1}\myvec{3 \\ 5}}{{\norm{\frac{1}{\sqrt{2}}\myvec{-1\\1}}}^2} =\sqrt{2}\\
\end{align}
Now, by \eqref{4}
\begin{align}
	\vec{u_2} = \frac{\myvec{3 \\ 5}- \sqrt{2}\frac{1}{\sqrt{2}}\myvec{-1 \\1}}{\norm{\myvec{3 \\ 5}- \sqrt{2}\frac{1}{\sqrt{2}}\myvec{-1 \\1}}} = 
	\frac{1}{\sqrt{2}}\myvec{1\\1}
\end{align}
From \eqref{5},
\begin{align}
	k_2 = \frac{1}{\sqrt{2}}\myvec{1&1} \myvec{3\\5} = \frac{8}{\sqrt{2}}
\end{align}
\textbf{Verification for $\vec{Q}^T\vec{Q}=I$}\\
Now,
\begin{align}
	\vec{Q} = 	\myvec{\frac{-1}{\sqrt{2}}& \frac{1}{\sqrt{2}}\\ \frac{1}{\sqrt{2}} & \frac{1}{\sqrt{2}}} 
\end{align}
Now, we observe that $\vec{Q}$$\vec{Q}^T$ = $\vec{Q}^T$$\vec{Q}$ = $\vec{I}$
\begin{multline}
	\myvec{\frac{-1}{\sqrt{2}}& \frac{1}{\sqrt{2}}\\ \frac{1}{\sqrt{2}} & \frac{1}{\sqrt{2}}} \myvec{\frac{-1}{\sqrt{2}}& \frac{1}{\sqrt{2}}\\ \frac{1}{\sqrt{2}} & \frac{1}{\sqrt{2}}} = \myvec{\frac{-1}{\sqrt{2}}& \frac{1}{\sqrt{2}}\\ \frac{1}{\sqrt{2}} & \frac{1}{\sqrt{2}}} \myvec{\frac{-1}{\sqrt{2}}& \frac{1}{\sqrt{2}}\\ \frac{1}{\sqrt{2}} & \frac{1}{\sqrt{2}}} = \\ 
	\myvec{1 & 0 \\ 0 & 1}
\end{multline}
Now, by \eqref{eq10} we can write matrix $\vec{V}$ as
\begin{multline}
\myvec{-1&3\\1&5}= \myvec{\frac{-1}{\sqrt{2}}& \frac{1}{\sqrt{2}}\\ \frac{1}{\sqrt{2}} & \frac{1}{\sqrt{2}}}\myvec{\sqrt{2}&\sqrt{2}\\0&\frac{8}{\sqrt{2}}}
\end{multline}
which is the required $\vec{QR}$ decomposition of $\vec{V}$.
